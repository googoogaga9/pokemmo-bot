# pokemmo-bot
Beats the game with no hands.

# Instructions for linux
## Create python virtual enviornment called .venv
python -m venv .venv

## Active .venv
source .venv/bin/activate

## Install dependencies
pip install -r requirements.txt

## Run program
python main.py

# Instructions for macos
python3 -m venv .venv
source .venv/bin/activate
pip install -r requirements.txt
python3 main.py
